\newcommand{\Instructions}[2]
{
  \begin{small}
    \begin{multicols}{2}
    \textbf{Instrucciones generales para las preguntas cerradas:}
    \begin{itemize}
      \item No se tendrán en cuenta anotaciones fuera de la tabla de respuestas. Utiliza MAYÚSCULAS.
      \item Todas las preguntas tienen el mismo valor.
      \item Hay una, y sólo una, respuesta correcta en cada pregunta.
      \item Responde solo a aquellas preguntas de las que estés seguro. Una respuesta incorrecta resta un tercio del valor de una respuesta acertada, pero una pregunta sin responder no resta puntos.
    \end{itemize}
    \vfill\columnbreak

    \textbf{Instrucciones generales para las preguntas abiertas:}
    \begin{itemize}
      \item Es necesario responder a la pregunta y justificar dicha respuesta.
      \item Todas las preguntas tienen el mismo valor.
    \end{itemize}
    \textbf{Puntuación:}
    \begin{itemize}
      \item La parte tipo test es un  #1\% de la nota.
      \item La parte de preguntas abiertas es un #2\% de la nota.
      \item Se necesita un mínimo de 3,5 en cada parte del examen para que hagan media.
    \end{itemize}
    \end{multicols}
  \end{small}
}
